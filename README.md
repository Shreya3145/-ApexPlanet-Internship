/* Make body a flex container */
body {

  min-height: 100vh;         /* full viewport height */
  margin: 0;                 /* remove default margin */
  background-color: #f0f0f0; /* optional background */
}

  text-align: center;
}

justify-content: center horizontally centers the container.
